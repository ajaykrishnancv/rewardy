# Rewardy Pro - Balanced Vision (Streamlined)

## Overview

**Rewardy Pro** is a focused homeschool management and motivation platform that combines timetable management, task tracking, and a gamified star-based reward economy. It's designed to be practical, achievable, and genuinely useful without unnecessary complexity.

---

## Philosophy

```
"Focused enough to ship, powerful enough to motivate."
```

### Design Principles
1. **Essential features done well** - Core functionality polished to perfection
2. **Parent control** - Full visibility and management for parents
3. **Fun without complexity** - Gamification that motivates without overwhelming
4. **Financial literacy** - Teaching saving, spending, and goal-setting through play

---

## Feature Summary

| Feature | What's Included |
|---------|-----------------|
| **User accounts** | Multi-parent (Admin) + Child |
| **Timetable** | Weekly templates, daily schedules, drag-and-drop |
| **Tasks** | Scheduled, habits, bonus tasks, quest-based goals |
| **Economy** | Stars + Gems, savings with interest, goal jars |
| **Rewards** | Configurable shop with approval workflow |
| **Gamification** | Quests (daily/weekly), achievements, streaks |
| **Skill Tracking** | Subject-wise progress and mastery levels |
| **Salary System** | Monthly salary with budget allocation (Phase 2+) |
| **Analytics** | Progress charts, weekly reports, insights |

### Database Schema
```sql
-- Core tables
CREATE TABLE families (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(100) NOT NULL,
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    family_id UUID REFERENCES families(id),
    email VARCHAR(255) UNIQUE,
    name VARCHAR(100) NOT NULL,
    role VARCHAR(20) NOT NULL, -- 'admin', 'child'
    preferences JSONB DEFAULT '{}',
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE child_profiles (
    user_id UUID PRIMARY KEY REFERENCES users(id),
    date_of_birth DATE,
    grade_level VARCHAR(20),
    learning_phase INTEGER DEFAULT 1,
    interests TEXT[],
    session_length_minutes INTEGER DEFAULT 25,
    break_frequency_minutes INTEGER DEFAULT 25
);

-- Schedule tables
CREATE TABLE schedule_templates (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    family_id UUID REFERENCES families(id),
    name VARCHAR(100),
    day_of_week INTEGER, -- 0-6, NULL for daily
    start_time TIME NOT NULL,
    end_time TIME NOT NULL,
    activity_name VARCHAR(100) NOT NULL,
    activity_type VARCHAR(30), -- 'study', 'break', 'routine', 'creative', 'physical'
    subject VARCHAR(50),
    star_value INTEGER DEFAULT 0,
    color VARCHAR(7),
    is_active BOOLEAN DEFAULT true,
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE daily_tasks (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    child_id UUID REFERENCES users(id),
    template_id UUID REFERENCES schedule_templates(id),
    date DATE NOT NULL,
    title VARCHAR(200) NOT NULL,
    description TEXT,
    task_type VARCHAR(30), -- 'scheduled', 'habit', 'bonus', 'quest'
    category VARCHAR(50),
    star_value INTEGER DEFAULT 0,
    status VARCHAR(20) DEFAULT 'pending', -- 'pending', 'submitted', 'approved', 'rejected'
    completed_at TIMESTAMP,
    approved_by UUID REFERENCES users(id),
    notes TEXT,
    created_at TIMESTAMP DEFAULT NOW()
);

-- Economy tables
CREATE TABLE currency_balances (
    user_id UUID REFERENCES users(id),
    currency_type VARCHAR(20), -- 'stars', 'gems'
    wallet_balance INTEGER DEFAULT 0,
    savings_balance INTEGER DEFAULT 0,
    total_earned INTEGER DEFAULT 0,
    total_spent INTEGER DEFAULT 0,
    PRIMARY KEY (user_id, currency_type)
);

CREATE TABLE savings_goals (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id),
    name VARCHAR(100) NOT NULL,
    target_amount INTEGER NOT NULL,
    current_amount INTEGER DEFAULT 0,
    currency_type VARCHAR(20) DEFAULT 'stars',
    is_completed BOOLEAN DEFAULT false,
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE transactions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id),
    currency_type VARCHAR(20) NOT NULL,
    amount INTEGER NOT NULL,
    transaction_type VARCHAR(30) NOT NULL, -- 'earn', 'spend', 'transfer', 'interest'
    category VARCHAR(50),
    description VARCHAR(200),
    balance_after INTEGER NOT NULL,
    related_task_id UUID REFERENCES daily_tasks(id),
    related_goal_id UUID REFERENCES savings_goals(id),
    created_by UUID REFERENCES users(id),
    created_at TIMESTAMP DEFAULT NOW()
);

-- Rewards tables
CREATE TABLE rewards (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    family_id UUID REFERENCES families(id),
    name VARCHAR(100) NOT NULL,
    description TEXT,
    star_cost INTEGER,
    gem_cost INTEGER,
    category VARCHAR(50), -- 'quick', 'experience', 'big', 'exclusive'
    image_url TEXT,
    is_active BOOLEAN DEFAULT true,
    auto_approve BOOLEAN DEFAULT false,
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE redemptions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id),
    reward_id UUID REFERENCES rewards(id),
    stars_spent INTEGER DEFAULT 0,
    gems_spent INTEGER DEFAULT 0,
    status VARCHAR(20) DEFAULT 'pending', -- 'pending', 'approved', 'rejected', 'fulfilled'
    approved_by UUID REFERENCES users(id),
    notes TEXT,
    created_at TIMESTAMP DEFAULT NOW()
);

-- Gamification tables
CREATE TABLE achievements (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(100) NOT NULL,
    description TEXT,
    category VARCHAR(50), -- 'streak', 'stars', 'savings', 'subject', 'special'
    badge_icon VARCHAR(50),
    criteria JSONB NOT NULL, -- {"type": "streak", "value": 7}
    gem_reward INTEGER DEFAULT 1
);

CREATE TABLE user_achievements (
    user_id UUID REFERENCES users(id),
    achievement_id UUID REFERENCES achievements(id),
    earned_at TIMESTAMP DEFAULT NOW(),
    PRIMARY KEY (user_id, achievement_id)
);

CREATE TABLE user_progress (
    user_id UUID PRIMARY KEY REFERENCES users(id),
    current_streak INTEGER DEFAULT 0,
    longest_streak INTEGER DEFAULT 0,
    last_activity_date DATE,
    updated_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE quests (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    family_id UUID REFERENCES families(id),
    quest_type VARCHAR(20), -- 'daily', 'weekly', 'subject', 'special'
    title VARCHAR(100) NOT NULL,
    description TEXT,
    criteria JSONB NOT NULL, -- {"type": "tasks_completed", "value": 5}
    star_reward INTEGER DEFAULT 0,
    gem_reward INTEGER DEFAULT 0,
    badge_reward UUID REFERENCES achievements(id),
    start_date DATE,
    end_date DATE,
    is_active BOOLEAN DEFAULT true
);

CREATE TABLE user_quests (
    user_id UUID REFERENCES users(id),
    quest_id UUID REFERENCES quests(id),
    progress INTEGER DEFAULT 0,
    status VARCHAR(20) DEFAULT 'active', -- 'active', 'completed', 'expired'
    completed_at TIMESTAMP,
    PRIMARY KEY (user_id, quest_id)
);

-- Subject progress
CREATE TABLE subject_progress (
    user_id UUID REFERENCES users(id),
    subject VARCHAR(50),
    level VARCHAR(20) DEFAULT 'beginner',
    points INTEGER DEFAULT 0,
    tasks_completed INTEGER DEFAULT 0,
    PRIMARY KEY (user_id, subject)
);

-- Salary system (Phase 2+)
CREATE TABLE salary_config (
    family_id UUID PRIMARY KEY REFERENCES families(id),
    is_enabled BOOLEAN DEFAULT false,
    base_amount DECIMAL(10,2),
    currency VARCHAR(10) DEFAULT 'INR',
    pay_frequency VARCHAR(20) DEFAULT 'monthly',
    save_percent INTEGER DEFAULT 50,
    spend_percent INTEGER DEFAULT 30,
    give_percent INTEGER DEFAULT 20
);

CREATE TABLE salary_records (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id),
    period_start DATE NOT NULL,
    period_end DATE NOT NULL,
    base_amount DECIMAL(10,2),
    bonus_amount DECIMAL(10,2) DEFAULT 0,
    total_amount DECIMAL(10,2),
    breakdown JSONB,
    status VARCHAR(20) DEFAULT 'pending',
    paid_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT NOW()
);

-- Create indexes for performance
CREATE INDEX idx_tasks_child_date ON daily_tasks(child_id, date);
CREATE INDEX idx_transactions_user ON transactions(user_id, created_at);
```
